    <script>
    const calendarContainer = document.getElementById('calendar-container');
    const monthSelector = document.getElementById('month-selector');
    const selectedEvent = document.getElementById('selected-event');

    const eventos = {
      '2025-11-05': 'Puerto Escondido üåä',
      '2025-11-06': 'Puerto Escondido üåä',
      '2025-11-07': 'Puerto Escondido üåä',
      '2025-11-08': 'Puerto Escondido üåä',
      '2025-12-18': 'Ixtapa üå¥',
      '2025-12-19': 'Ixtapa üå¥',
      '2025-12-20': 'Ixtapa üå¥',
      '2025-12-21': 'Ixtapa üå¥',
      '2025-12-31': 'Acapulco üéÜ'
    };

    const daysOfWeek = ['Dom', 'Lun', 'Mar', 'Mi√©', 'Jue', 'Vie', 'S√°b'];

    function generarCalendario(year, month) {
      calendarContainer.innerHTML = '';
      selectedEvent.innerText = '';
      const firstDay = new Date(year, month - 1, 1).getDay();
      const totalDays = new Date(year, month, 0).getDate();

      daysOfWeek.forEach(d => {
        const div = document.createElement('div');
        div.className = 'day-name';
        div.textContent = d;
        calendarContainer.appendChild(div);
      });

      for (let i = 0; i < firstDay; i++) {
        calendarContainer.appendChild(document.createElement('div'));
      }

      for (let d = 1; d <= totalDays; d++) {
        const dateKey = `${year}-${String(month).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
        const div = document.createElement('div');
        div.className = 'day';
        div.textContent = d;

        if (eventos[dateKey]) {
          div.classList.add('event-day');
          div.title = eventos[dateKey];
        }

        div.onclick = () => {
          div.classList.toggle('selected');
          mostrarEventosSeleccionados();
        };

        calendarContainer.appendChild(div);
      }
    }

    function mostrarEventosSeleccionados() {
      const seleccionados = Array.from(document.querySelectorAll('.day.selected'))
        .map(d => d.title)
        .filter(Boolean);
      selectedEvent.textContent = seleccionados.length > 0 ?
        `Seleccionaste: ${[...new Set(seleccionados)].join(', ')}` : '';
    }

    generarCalendario(2025, 11);
    monthSelector.onchange = e => {
      const [year, month] = e.target.value.split('-');
      generarCalendario(+year, +month);
    };
